```python
from app.services.csv_sampling import detect_delimiter
import polars as pl

delimiter_edi = detect_delimiter(edi_csv_path, encoding="utf8")
delimiter_preisliste = detect_delimiter(preisliste_csv_path, encoding="utf8")

df_edi = pl.read_csv(edi_csv_path, separator=delimiter_edi, encoding="utf8", truncate_ragged_lines=True)
df_preisliste = pl.read_csv(preisliste_csv_path, separator=delimiter_preisliste, encoding="utf8", truncate_ragged_lines=True)
```

**Left Join on Artikelnummer** (D-01 from CONTEXT):
```python
merged_df = df_edi.join(df_preisliste, on="Artikelnummer", how="left", suffix="_preisliste")
```

**Field-Specific Conflict Resolution** (D-05, D-06, D-07 from CONTEXT):
- Preisliste fields to use: preis, menge1, menge2, menge3, menge4, menge5
- For each product, build data dict with these rules:
  * Take all EDI fields first
  * Override with Preisliste price fields if present (not null)
  * Track source for each field in sources dict

**Handle Missing Data** (D-09, D-10 from CONTEXT):
- Use None for null values (not empty strings)
- products without price match keep all EDI data with price fields as None

**Return** MergeResult with list of MergedProduct objects.

```python
import pytest
import polars as pl
from pathlib import Path
from app.services.csv_merge import merge_csv_data
from app.models.merge import MergeResult, MergedProduct

@pytest.fixture
def sample_edi_csv(tmp_path):
    # Create minimal EDI CSV with semicolon delimiter
    csv_content = "Artikel nummer;Bezeichnung1;Gewicht\n210100125;VERSANDTASCHE;0.028\n210100225;DVD TASCHE;0.045"
    csv_file = tmp_path / "edi.csv"
    csv_file.write_text(csv_content, encoding="utf-8")
    return csv_file

@pytest.fixture
def sample_preisliste_csv(tmp_path):
    # Create minimal Preisliste CSV with semicolon delimiter
    csv_content = "Artikelnummer;preis;menge1\n210100125;1.50;100\n999999999;2.00;50"
    csv_file = tmp_path / "preisliste.csv"
    csv_file.write_text(csv_content, encoding="utf-8")
    return csv_file

def test_merge_basic(sample_edi_csv, sample_preisliste_csv):
    result = merge_csv_data(sample_edi_csv, sample_preisliste_csv)
    assert result.total_products == 2
    assert result.matched == 1  # 210100125 in both
    assert result.edi_only == 1  # 210100225 only in EDI

def test_field_priority(sample_edi_csv, sample_preisliste_csv):
    result = merge_csv_data(sample_edi_csv, sample_preisliste_csv)
    product = next(p for p in result.products if p.artikelnummer == "210100125")
    assert product.data["preis"] == 1.50  # From Preisliste
    assert product.data["gewicht"] == 0.028  # From EDI
    assert product.sources["preis"] == "preisliste"
    assert product.sources["gewicht"] == "edi_export"

def test_missing_price(sample_edi_csv, sample_preisliste_csv):
    result = merge_csv_data(sample_edi_csv, sample_preisliste_csv)
    product = next(p for p in result.products if p.artikelnummer == "210100225")
    assert product.data["preis"] is None  # No match in Preisliste
    assert product.sources["preis"] is None
```
